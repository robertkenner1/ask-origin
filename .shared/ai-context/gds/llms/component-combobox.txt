# Combobox

**Full path**: GDS/Combobox

**Import**:
```typescript
import { Combobox } from '@grammarly/design-system';
```

## Examples

### Default

```tsx
(
    <Combobox label="Favorite animal">
      <ComboboxItem value="red panda" textValue="Red Panda">
        Red Panda
      </ComboboxItem>
      <ComboboxItem value="cat">Cat</ComboboxItem>
      <ComboboxItem value="dog">Dog</ComboboxItem>
    </Combobox>
  )
```

### WithCustomValuesAllowed

```tsx
(
    <Combobox label="Favorite animal" allowsCustomValue={true}>
      <ComboboxItem value="red panda">Red Panda</ComboboxItem>
      <ComboboxItem value="cat">Cat</ComboboxItem>
      <ComboboxItem value="dog">Dog</ComboboxItem>
    </Combobox>
  )
```

### WithHelperText

```tsx
(
    <Combobox
      label="Favorite animal"
      helperMessage="Pick the one you want as a pet"
      defaultSelectedItem="cat"
    >
      <ComboboxItem value="red panda">Red Panda</ComboboxItem>
      <ComboboxItem value="cat">Cat</ComboboxItem>
      <ComboboxItem value="dog">Dog</ComboboxItem>
    </Combobox>
  )
```

### WithNoResults

```tsx
(
    <Combobox label="Favorite animal" helperMessage="Pick the one you want as a pet">
      <ComboboxItem value="red panda">Red Panda</ComboboxItem>
      <ComboboxItem value="cat">Cat</ComboboxItem>
      <ComboboxItem value="dog">Dog</ComboboxItem>
    </Combobox>
  )
```

### WithNoResultsCustom

```tsx
(
    <Combobox
      label="Favorite animal"
      noResultsMessage="No animals found"
      helperMessage="Pick the one you want as a pet"
    >
      <ComboboxItem value="red panda">Red Panda</ComboboxItem>
      <ComboboxItem value="cat">Cat</ComboboxItem>
      <ComboboxItem value="dog">Dog</ComboboxItem>
    </Combobox>
  )
```
