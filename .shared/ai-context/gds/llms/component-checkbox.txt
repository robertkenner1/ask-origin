### ariaDescribedByAndHelperTextBothSet

```tsx
{
    return (
      <>
        <Checkbox
          aria-describedby="remember-me-desc"
          helperMessage="Browser will save your info for 7 days"
        >
          Remember me
        </Checkbox>
        <Text as="p" id="remember-me-desc" variant="text-small">
          Additional text about this checkbox. Aria-describedby should point to the helperMessage
          since it was provided.
        </Text>
      </>
    );
  }
```
