import ComponentPlayground, { createCodeMonitor, formatCode } from "./_components/comp_playground";
import DogBoneIcon from "/static/img/onboarding/dog-bone.svg";
import useBaseUrl from "@docusaurus/useBaseUrl";
import { Button, Flex } from "@grammarly/design-system"
import * as prettier from "prettier/standalone";
import * as React from "react";

# Onboarding demo

export const OnboardingCodePlayground = () => {
    const defaultCode = `function Example() {
  return (
  <Flex
      width={365}
      height={365}
      borderRadius={3}
      elevation="low"
      direction="column"
      style={{ overflow: "hidden" }}
    >
      <Flex height={280}>
        <img src={useBaseUrl("/img/onboarding/fen.png")} alt="" />
      </Flex>
      <Flex padding={4}  align="start">
        <Flex direction="column">
            Text
        </Flex>
        <Flex align="center" gap={1}>
          <IconButton
            variant="tertiary"
            accessibilityLabel=""
            icon={DogBoneIcon}
          />
        </Flex>
      </Flex>
    </Flex>)}`;
    const [formattedCode, setFormattedCode] = React.useState(defaultCode);
    let transformedCode = null;

    const monitorCode = createCodeMonitor((code) => {
        transformedCode = code
    });

    return (
        <Flex direction="column" gap={1}>
            <Button
                style={{ alignSelf: "end" }}
                text="Format code"
                variant="tertiary"
                onClick={() => setFormattedCode(formatCode(transformedCode))}
            />
            <ComponentPlayground
                monitorCode={monitorCode}
                autoStart={true}
                scope={{ DogBoneIcon, useBaseUrl }}
                style={{ minWidth: "100%", minHeight: "400px" }}
                layout="grid"
                code={formattedCode}
            />
        </Flex>
    );
};

<OnboardingCodePlayground/>
