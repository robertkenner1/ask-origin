import Link from "@docusaurus/Link";
import { componentStatusData, getStatusText } from "../_components/component_status_data";
import { ComponentStatusItem } from "../_components/component_status_row";
import { Flex, Tag, Text } from "@grammarly/design-system";

export function getComponentLink(component) {
  const componentLocation = component.type === "utility" ? "utilities" : "components";
  const componentName = component.name.toLowerCase().replace(/ /g, "-");

  if (component.subgroup) {
    return (`/${componentLocation}/${component.subgroup}/${componentName}`);
  }
  return (`/${componentLocation}/${componentName}`);
}

Components are available in both Figma and React. The status of each component is tracked in the table below.

You can also interact with each component in the [GDS Storybook](https://uifoundation.gpages.io/grammarly-design-system/) or the [Component Playground](/components/component-playground).
