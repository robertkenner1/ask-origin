export const CodePlayground = () => {
  const [playgroundDisplay, setPlaygroundDisplay] = React.useState("row");
  const [playgroundExample, setPlaygroundExample] = React.useState("kitchenSink");
  const [currentCode, setCurrentCode] = React.useState("");
  const [showCopyToast, setShowCopyToast] = React.useState(false);
  let transformedCode = null;

const monitorCode = createCodeMonitor((code) => {
transformedCode = code
});

const TOAST_TIMEOUT_MS = 3000;
let toastTimeout;

React.useEffect(() => {
if (window.location.hash.length > 1) {
const code = LZString.decompressFromEncodedURIComponent(window.location.hash.substring(1))
PlaygroundExamples['code'] = code
setCurrentCode(code)
setPlaygroundExample('code')
} else {
setCurrentCode(PlaygroundExamples[playgroundExample])
}
}, []);

function onExampleChange(value) {
setPlaygroundExample(value)

    window.location.hash = ''
    setCurrentCode(PlaygroundExamples[value])

}

function onToastClose() {
setShowCopyToast(false)
toastTimeout = undefined
}

async function copyTextToClipboard(text) {
if ("clipboard" in navigator) {
return await navigator.clipboard.writeText(text);
} else {
return document.execCommand("copy", true, text);
}
}

async function onShareClick(value) {
setCurrentCode(transformedCode)
window.location.hash = LZString.compressToEncodedURIComponent(transformedCode);
await copyTextToClipboard(window.location);

    toastTimeout = setTimeout(onToastClose, TOAST_TIMEOUT_MS)
    setShowCopyToast(true)

}

return (
<Flex borderRadius={2} direction="column" borderColor="base-subdued">
{showCopyToast ?
<Toast
text="Link copied to clipboard."
onClose={onToastClose}
variant="success"
style={
{
position: 'fixed',
bottom: '1rem',
right: '1rem',
zIndex: 'var(--z-index-tooltip)'
}} />
: null}
<Flex
bgColor="background-base-subdued"
borderRadius={2}
padding={4}
gap={6}
align="start"
style={{
          borderBottom: "1px solid var(--color-border-base-subdued)",
          borderBottomLeftRadius: "0px",
          borderBottomRightRadius: "0px"
        }}>
<Flex width="30%">
<Select value={playgroundExample} label="Starter templates" onChange={onExampleChange}>
<Select.Option value="kitchenSink" label="Kitchen Sink" />
<Select.Option value="card" label="Card" />
<Select.Option value="onboardingDemo" label="Granimals Card" />
<Select.Option value="empty" label="Empty Example" />
{'code' in PlaygroundExamples ? <Select.Option value="code" label="Your Code" /> : null}
</Select>
</Flex>
<RadioGroup
          legend="Display settings"
          layout="row"
          onChange={setPlaygroundDisplay}
          defaultValue="row"
        >
<RadioButton value="row">Side by side</RadioButton>
<RadioButton value="column">Code on the bottom</RadioButton>
</RadioGroup>
<div style={{ flexGrow: 1 }}></div>
<Button iconStart={InterfaceCopyIcon} text="Copy link to code" onClick={onShareClick} style={{ alignSelf: "center" }} />
</Flex>
<ComponentPlayground
monitorCode={monitorCode}
scope={{ DogBoneIcon, useBaseUrl }}
autoStart={true}
style={{ minWidth: "900px", minHeight: "400px", border: "none" }}
layout={playgroundDisplay}
code={currentCode} />
</Flex>
)
}
